    listen 80;                         # Порт, который слушает Nginx снаружи
    server_name example.com;           # Ваше доменное имя или IP-адрес

        proxy_pass http://172.17.0.2:80; # Внутренний адрес и порт приложения

        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
